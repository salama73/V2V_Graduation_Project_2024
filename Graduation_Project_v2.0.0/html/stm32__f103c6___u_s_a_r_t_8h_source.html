61// @ ref USART_MODE_Define
62/*
63Bit 2 RE: Receiver enable
64This bit enables the receiver. It is set and cleared by software.
650: Receiver is disabled
661: Receiver is enabled and begins searching for a start bit
67
68Bit 3 TE: Transmitter enable
69This bit enables the transmitter. It is set and cleared by software.
700: Transmitter is disabled
711: Transmitter is enabled
72 */
73#define USART_MODE_TX (uint32)(1<<2)
74#define USART_MODE_RX (uint32)(1<<3)
75#define USART_MODE_TX_RX (uint32)(1<<2 | 1<<3 )

80 Bit 12 M: Word length

831: 1 Start bit, 9 Data bits, n Stop bit
84 */

156#define USARTDIV_MUL100( PCLK , BAUD) ( uint32 )( ( 25 * PCLK ) / ( 4 * BAUD ) )
